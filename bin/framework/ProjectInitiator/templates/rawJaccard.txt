

const calculate = (arr1, arr2) => {
    const set1 = new Set(arr1);
    const set2 = new Set(arr2);
    
    const intersection = [...arr1].filter(x => set2.has(x));
    const union = new Set([...set1], [...set2]);

    return intersection.length / union.size;
}

module.exports = {
    calculate
};
